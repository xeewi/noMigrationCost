**Section B — Alpine appState() function:**

```javascript
function appState() {
  return {
    // Team state (per D-01: fixed 4-slot grid)
    team: SENIORITY_LEVELS.map(level => ({
      seniority: level.label,
      headcount: 0,
      hourlyRate: level.defaultRate
    })),

    // Feature sizing state (per D-05, D-06, D-07)
    sizingMode: 'sp',          // 'sp' | 'hours'
    storyPoints: 0,
    velocity: 30,              // SP/sprint default (docs section 1.2)
    sprintWeeks: 2,            // default 2-week sprints
    directHours: 0,
    directDays: 0,
    directUnit: 'hours',       // 'hours' | 'days'

    // Time horizon (per D-09)
    horizonYears: 3,           // default 3 years

    // Computed: team average hourly cost (per D-04, TEAM-05)
    get teamAverageRate() {
      const totalHeadcount = this.team.reduce((s, r) => s + r.headcount, 0);
      if (totalHeadcount === 0) return 0;
      const weightedSum = this.team.reduce((s, r) => s + r.headcount * r.hourlyRate, 0);
      return weightedSum / totalHeadcount;
    },

    // Computed: estimated development hours (per D-08, SIZE-05)
    get devHours() {
      if (this.sizingMode === 'sp') {
        return spToHours(this.storyPoints, this.velocity, this.sprintWeeks);
      } else {
        if (this.directUnit === 'days') return this.directDays * FORMULA_DEFAULTS.hoursPerDay;
        return this.directHours;
      }
    },

    // Computed: total standalone cost (per COST-01)
    get standaloneCost() {
      return computeStandaloneCost(this.devHours, this.teamAverageRate, this.horizonYears);
    },

    // Computed: breakdown rows (per COST-02, D-11)
    get costBreakdown() {
      return computeBreakdown(this.devHours, this.teamAverageRate, this.horizonYears);
    },

    // Computed: is the output ready to show? (per UI-SPEC States)
    get hasValidInputs() {
      return this.teamAverageRate > 0 && this.devHours > 0;
    }
  };
}
```
