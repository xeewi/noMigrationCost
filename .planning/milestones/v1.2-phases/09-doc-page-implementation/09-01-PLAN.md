2. **toSlug function** — matches github-slugger behavior for ASCII headings:
   ```typescript
   function toSlug(text: string): string {
     return text
       .toLowerCase()
       .replace(/[^\w\s-]/g, '')
       .replace(/\s+/g, '-')
       .replace(/-+/g, '-')
       .replace(/^-|-$/g, '');
   }
   ```

3. **extractHeadings function** — parses raw markdown string, extracts h2 and h3 headings:
   ```typescript
   function extractHeadings(markdown: string): HeadingEntry[] {
     const entries: HeadingEntry[] = [];
     const counts: Record<string, number> = {};
     for (const line of markdown.split('\n')) {
       const m = line.match(/^(#{2,3})\s+(.+)$/);
       if (!m) continue;
       const level = m[1].length as 2 | 3;
       const text = m[2].trim();
       const base = toSlug(text);
       const count = counts[base] ?? 0;
       counts[base] = count + 1;
       const slug = count === 0 ? base : `${base}-${count}`;
       entries.push({ id: `doc-${slug}`, text, level });
     }
     return entries;
   }
   ```
